let isMouseDown = false;

document.addEventListener('mousedown', function() {
    isMouseDown = true;
});

document.addEventListener('mouseup', function() {
    isMouseDown = false;
});

document.addEventListener('mousemove', function(e) {
    if (isMouseDown) {
        var el = document.querySelector('.ver-perExp');
        var x = e.clientX;
        var y = e.clientY;

        // Ajusta los valores para que el contenedor no se desplace completamente fuera de la pantalla
        var offsetX = -el.offsetWidth / 2;
        var offsetY = -el.offsetHeight / 2;

        el.style.left = (x + offsetX) + 'px';
        el.style.top = (y + offsetY) + 'px';
    }
});
